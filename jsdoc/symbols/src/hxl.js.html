 50 /**
 51  * Wrap a JavaScript array as a HXL dataset.
 52  *
 53  * @param {array} rawData an array of arrays of raw strings to wrap
 54  * @return a new {@link hxl.classes.Dataset}

 56 hxl.wrap = function (rawData) {
 57     return new hxl.classes.Dataset(rawData);

 60 /**
 61  * Load a remote HXL dataset asynchronously.
 62  *
 63  * The callback takes a single argument, which is
 64  * the HXL data source (when successfully loaded).
 65  *
 66  * @param {string} url The URL of the HXL dataset to load.
 67  * @param {function} callback The function to call when loaded.
 68  */
 69 hxl.load = function (url, callback) {
 70     if (typeof(Papa) != 'undefined' && typeof(Papa.parse) != 'undefined') {
 71         Papa.parse(url, {
 72             download: true,
 73             skipEmptyLines: true,
 74             complete: function (result) {
 75                 callback(hxl.wrap(result.data));
 76             },
 77             error: function (result) {
 78                 throw new Error(result.errors.join("\n"));
 79             }
 80         });
 81     } else {
 82         throw Error("No CSV parser available (tried Papa.parse)");
 83     }
 84 };
 85 
 86 /** 
 87 * Load a HXL dataset from a string
 88 */
 89 hxl.parseString = function(string) {
 90     return hxl.wrap(Papa.parse(string).data);
 91 }

 93 /**
 94  * Normalise case and whitespace in a string.
 95  *
 96  * @param {string} value The string value to normalise.
 97  * @return {string} the string with case and whitespace normalised.
 98  */
 99 hxl.norm = function (value) {
100     if (value) {
101         return value.toString().trim().replace(/\s+/, ' ').toLowerCase();
102     } else {
103         return null;
104     }
105 };

168 /**
169  * Get an iterator to read through the rows of a HXL dataset.
170  *
171  * The iterator will have a next() method to read each row in sequence, and
172  * will return null once the rows are exhausted.
173  *
174  * <pre>
175  * var row;
176  * var iterator = data.iterator();
177  * while (row = iterator.next()) {
178  *   // do something with the row
179  * }
180  * </pre>
181  *
182  * The {@link #each} or {@link #forEach} method provides a
183  * more-elegant way of iterating through rows, but this method makes
184  * it easier to break out of the loop early.
185  *
186  * @function
187  * @return {object} An iterator object with a .next() method.
188  * @see #getRows
189  * @see #each
190  * @see #forEach
191  */
192 hxl.classes.Source.prototype.iterator = undefined;

194 /**
195  * Get an array of row objects.
196  *
197  * This method might be highly inefficient, depending on the
198  * implementation in the derived class. Normally, it's best
199  * to go through the rows using an iterator.
200  *
201  * <pre>
202  * var rows = data.getRows();
203  * </pre>
204  *
205  * It is usually easier to use the {@link #rows} property, which is an
206  * alias to this method:
207  *
208  * <pre>
209  * var rows = data.rows;
210  * </pre>
211  *
212  * @return {array} An array of {@link hxl.classes.Row} objects.
213  * @see #rows
214  * @see #iterator
215  * @see #getColumns
216  */
217 hxl.classes.Source.prototype.getRows = function () {
218     var row;
219     var rows = [];
220     var iterator = this.iterator();
221     while (row = iterator.next()) {
222         rows.push(row);
223     }
224     return rows;
225 }

307 /**
308  * Get the minimum value for a column
309  *
310  * Uses a < comparison, ignoring empty cells. This method is
311  * especially useful for setting ranges in a chart or other
312  * visualisation.
313  *
314  * <pre>
315  * var minAffected = data.getMin('#affected');
316  * </pre>
317  *
318  * @param {string} pattern The tag pattern for the column(s) to
319  * use. See {@link hxl.classes.Pattern} for details.
320  * @return {string} The lowest value in the column.
321  * @see #getMax
322  */
323 hxl.classes.Source.prototype.getMin = function(pattern) {
324     var min = null;
325     var row, value;
326     var iterator = this.iterator();
327     pattern = hxl.classes.Pattern.parse(pattern); // more efficient to precompile
328     while (row = iterator.next()) {
329         value = row.get(pattern);
330         if (value !== null) {
331             num = parseFloat(value);
332             if (num !== NaN) {
333                 if (min === null || value < min) {
334                     min = value;
335                 }
336             }
337         }
338     }
339     return min;
340 }

347  *

349  * var maxAffected = data.getMax('#affected');
350  * </pre>
351  *
352  * @param {string} pattern The tag pattern for the column(s) to
353  * use. See {@link hxl.classes.Pattern} for details.
354  * @return {string} the highest value in the column.
355  * @see #getMin
356  */

376 /**
377  * Get a list of unique values for a tag
378  * 
379  * Uses a map to store unique values.
380  *
381  * <pre>
382  * var sectorList = data.getValues('#sector');
383  * </pre>
384  *
385  * @param {hxl.classes.Pattern} pattern A tag pattern to match for the column(s).
386  * @return {array} A list of unique values.
387  */
388 hxl.classes.Source.prototype.getValues = function(pattern) {
389     var row;
390     var iterator = this.iterator();
391     var value_map = {};
392 
393     pattern = hxl.classes.Pattern.parse(pattern); // more efficient to precompile
394     while (row = iterator.next()) {
395         value_map[row.get(pattern)] = true;
396     }
397     return Object.keys(value_map);
398 }

400 /**
401  * Check if a dataset contains at least one column matching a pattern.
402  *
403  * <pre>
404  * if (data.hasColumn('#meta+count')) {
405  *   draw_my_graph(data);
406  * }
407  * </pre>
408  *
409  * @param {hxl.classes.Pattern} pattern A tag pattern to match for the column(s).
410  * @return {boolean} true if there's a match, false otherwise.
411  * @see #getColumns
412  */
413 hxl.classes.Source.prototype.hasColumn = function (pattern) {
414     var cols =this.getColumns();
415     pattern = hxl.classes.Pattern.parse(pattern); // more efficient to precompile
416     for (var i = 0; i < cols.length; i++) {
417         if (pattern.match(cols[i])) {
418             return true;
419         }
420     }
421     return false;
422 }
423 
424 /**
425  * Get a list of indices for columns matching a tag pattern.
426  */
427 hxl.classes.Source.prototype.getMatchingColumns = function(pattern) {
428     var result = [];
429     pattern = hxl.classes.Pattern.parse(pattern); // more efficient to precompile
430     this.getColumns().forEach(function (col) {
431         if (pattern.match(col)) {
432             result.push(col);
433         }
434     });
435     return result;
436 }

438 /**
439  * Fire a callback on each row of data.
440  *
441  * The callback has the form
442  *
443  * <pre>
444  * function (row, source, rowNumber) {}
445  * </pre>
446  *
447  * (Often, the callback will need to bother with only the first parameter,
448  * so function (row) {} will be more typical).
449  *
450  * <pre>
451  * var sectors = {};
452  * data.each(function(row) {
453  *   sectors[row.getValue('sector')] = true;
454  * });
455  * </pre>
456  *
457  * @param {function} callback Callback function for each row of data.
458  * @return {int} The number of rows processed.
459  * @see #iterator
460  */
461 hxl.classes.Source.prototype.each = function(callback) {
462     var row, rowNumber = 0, iterator = this.iterator();
463     while (row = iterator.next()) {
464         callback(row, this, rowNumber++);
465     }
466     return rowNumber;
467 }

483 /**
484  * Test if a tag pattern points mainly to numbers.
485  *
486  * This method is useful for trying to guess which columns contain numeric
487  * values that can be graphed:
488  *
489  * <pre>
490  * if (data.isNumbery('#affected')) {
491  *   min = data.getMin('#affected');
492  * }
493  * </pre>
494  *
495  * @param {hxl.classes.Pattern} pattern A tag pattern to match for the column(s).
496  * @return {boolean} true if at least 90% of the non-null values are numeric.
497  */
498 hxl.classes.Source.prototype.isNumbery = function(pattern) {
499     var total_seen = 0;
500     var numeric_seen = 0;
501     pattern = hxl.classes.Pattern.parse(pattern); // more efficient to precompile
502     this.rows.forEach(function (row) {
503         var value = row.get(pattern);
504         if (value) {
505             total_seen++;
506             if (!isNaN(value)) {
507                 numeric_seen++;
508             }
509         }
510     });
511     return (total_seen > 0 && (numeric_seen/total_seen >= 0.9));;
512 }

681 /**
682  * An original HXL dataset (including the raw data)
683  * Derived from hxl.classes.Source
684  * @constructor
685  * @augments hxl.classes.Source
686  * @param {array} rawData An array of arrays of raw strings (tabular).
687  */
688 hxl.classes.Dataset = function (rawData) {
689     hxl.classes.Source.call(this);
690     this._rawData = rawData;
691     this._tagRowIndex = null;
692     this._savedColumns = null;
693 }
694 
695 hxl.classes.Dataset.prototype = Object.create(hxl.classes.Source.prototype);
696 hxl.classes.Dataset.prototype.constructor = hxl.classes.Dataset;
697 
698 /**
699  * Get an array of column definitions.
700  */
701 hxl.classes.Dataset.prototype.getColumns = function() {
702     var cols, tags_index, tagspec, header;
703     if (this._savedColumns == null) {
704         cols = [];
705         tags_index = this._getTagRowIndex();
706         if (tags_index > -1) {
707             for (var i = 0; i < this._rawData[tags_index].length; i++) {
708                 tagspec = this._rawData[tags_index][i];
709                 header = null;
710                 if (tags_index > 0) {
711                     header = this._rawData[tags_index-1][i];
712                 }
713                 if (tagspec.match(/^\s*#.*/)) {
714                     cols.push(hxl.classes.Column.parse(tagspec, header));
715                 } else {
716                     cols.push(null);
717                 }
718             }
719             this._savedColumns = cols;
720         } else {
721             throw "No HXL hashtag row found.";
722         }
723     }
724     return this._savedColumns;
725 }
726 
727 /**
728  * Get an iterator through all the rows in the dataset.
729  */
730 hxl.classes.Dataset.prototype.iterator = function() {
731     var index = this._getTagRowIndex() + 1;
732     var columns = this.columns;
733     var rawData = this._rawData;
734     return {
735         next: function() {
736             if (index < rawData.length) {
737                 return new hxl.classes.Row(rawData[index++], columns);
738             } else {
739                 return null;
740             }
741         }
742     };
743 }

745 /**
746  * Get the index of the tag row.
747  */
748 hxl.classes.Dataset.prototype._getTagRowIndex = function() {
749     var i;
750     if (this._tagRowIndex == null) {
751         for (i = 0; i < 25 && i < this._rawData.length; i++) {
752             if (this._isTagRow(this._rawData[i])) {

755             }

757         throw "No HXL tag row found."
758     } else {

760     }

846 /**
847  * A compiled pattern for matching HXL columns.
848  *
849  * <p>You should not normally call this constructor directly. All
850  * functions that use a pattern compile it as needed, so you can stick
851  * with strings.  If you want to precompile a pattern, use the static
852  * {@link #parse} method.</p>
853  *
854  * <p>The other useful method in this class is {@link #match}, with
855  * tests the pattern against a {@link hxl.classes.Column}.</p>
856  *
857  * <p>A pattern specification looks like a normal HXL hashtag and
858  * attributes, except that it can contain "-" attributes as well as
859  * "+" attributes. For example, the following pattern matches any
860  * column with the tag "#org", the attribute "+funder", and *not* the
861  * attribute "+impl" (any other attributes are ignored):</p>
862  *
863  * <pre>
864  * #org+funder-impl
865  * </pre>
866  *
867  * <p>More examples:</p>
868  *
869  * <p><b>#adm1</b> will match the following: #adm1, #adm1+code, #adm1+fr,
870  * #adm1+name, etc.</p>
871  *
872  * <p><b>#adm1-code</b> will match #adm1, #adm1+fr, and #adm1+name,
873  * but *not* #adm1+code (because the +code attribute is explicitly
874  * forbidden).</p>
875  *
876  * <p><b>#adm1+name-fr</b> will match #adm1+name or #adm1+name+en, but
877  * not #adm1+name+fr</p>
878  *
879  * @constructor
880  * @param {string} tag The basic tag to use in the pattern, with the
881  * leading '#'
882  * @param {array} include_attributes A (possibly-empty) list of
883  * attribute names that must be present, without the leading '+'
884  * @param {array} exclude_attributes A (possibly-empty) list of
885  * attribute names that must *not* be present, without the leading '+'
886  * @see hxl.classes.Column
887  */
888 hxl.classes.Pattern = function (tag, include_attributes, exclude_attributes) {
889     this.tag = tag;
890     this.include_attributes = include_attributes;
891     this.exclude_attributes = exclude_attributes;
892 }
893 
894 /**
895  * Test if a column matches this pattern.
896  *
897  * <pre>
898  * data.columns.forEach(function(col) {
899  *   if (pattern.match(col)) {
900  *     console.log("Found a match: " + col.displayTag);
901  *   }
902  * });
903  * </pre>
904  *
905  * This method is used heavily in the filter classes.
906  *
907  * @param {hxl.classes.Column} column The column to test.
908  * @return {boolean} true on match, false otherwise.
909  */
910 hxl.classes.Pattern.prototype.match = function(column) {
911     var attribute, i;
912 
913     // tags must match
914     if (this.tag != column.tag) {
915         return false;
916     }
917 
918     // include attributes must be present
919     for (i = 0; i < this.include_attributes.length; i++) {
920         attribute = this.include_attributes[i];
921         if (column.attributes.indexOf(attribute) < 0) {
922             return false;
923         }
924     }
925 
926     // exclude attributes must not be present
927     for (i = 0; i < this.exclude_attributes.length; i++) {
928         attribute = this.exclude_attributes[i];
929         if (column.attributes.indexOf(attribute) > -1) {
930             return false;
931         }
932     }
933 
934     return true;
935 }
936 
937 /**
938  * Parse a string into a tag pattern.
939  *
940  * <pre>
941  * var pattern = hxl.classes.Pattern.parse('#org+funder-impl');
942  * </pre>
943  *
944  * It is safe to pass an already-compiled {@link hxl.classes.Pattern}
945  * to this method; it will simple be returned as-is.
946  *
947  * @param {string} pattern a tag-pattern string, like "#org+funder-impl"
948  * @param useException (optional) throw an exception on failure.
949  * @return a hxl.classes.Pattern, or null if parsing fails (and useException is false).
950  */
951 hxl.classes.Pattern.parse = function(pattern, useException) {
952     var result, include_attributes, exclude_attributes, attribute_specs, i;
953     if (pattern instanceof hxl.classes.Pattern) {
954         // If this is already parsed, then just return it.
955         return pattern;
956     } else if (!pattern) {
957         if (useException) {
958             throw new Error("No tag pattern provided");
959         } else {
960             return null;
961         }
962     } else {
963         result = String(pattern).match(/^\s*#?([A-Za-z][A-Za-z0-9_]*)((?:\s*[+-][A-Za-z][A-Za-z0-9_]*)*)\s*$/);
964         if (result) {
965             include_attributes = [];
966             exclude_attributes = [];
967             attribute_specs = result[2].split(/\s*([+-])/).filter(function(item) { return item; });
968             for (i = 0; i < attribute_specs.length; i += 2) {
969                 if (attribute_specs[i] == "+") {
970                     include_attributes.push(attribute_specs[i+1]);
971                 } else {
972                     exclude_attributes.push(attribute_specs[i+1]);
973                 }
974             }
975             return new hxl.classes.Pattern('#' + result[1], include_attributes, exclude_attributes);
976         } else if (useException) {
977             throw "Bad tag pattern: " + pattern;
978         } else {
979             hxl.log("Bad tag pattern: " + pattern);
980             return null;
981         }
982     }
983 }
984 
985 hxl.classes.Pattern.toString = function() {
986     var s = this.tag;
987     if (this.include_tags) {
988         s += "+" + this.include_tags.join("+");
989     }
990     if (this.exclude_tags) {
991         s += "-" + this.exclude_tags.join("-");
992     }
993     return s;
994 }

1093 /**
1094  * HXL filter class to select rows from a source.
1095  *
1096  * Usage:
1097  *
1098  * // select all rows where #adm1 is the Coastal Region
1099  * // *or* the population is greater than 1,000
1100  * var filter = new hxl.classes.RowFilter(source,
1101  *   { pattern: '#adm1', test: 'Coastal Region' },
1102  *   { pattern: '#people_num', test: function(v) { return v > 1000; } }
1103  * ]);
1104  *
1105  * Predicates are always "OR"'d together. If you need
1106  * a logical "AND", then chain another select filter.
1107  *
1108  * @constructor
1109  * @param source the hxl.classes.Source
1110  * @param predicates a list of predicates, each of 
1111  * has a "test" property (and optionally, a "pattern" property).
1112  */
1113 hxl.classes.RowFilter = function (source, predicates, invert) {
1114     hxl.classes.BaseFilter.call(this, source);
1115     this.predicates = this._compilePredicates(predicates);
1116     this.invert = invert;
1117 }
1118 
1119 hxl.classes.RowFilter.prototype = Object.create(hxl.classes.BaseFilter.prototype);
1120 hxl.classes.RowFilter.prototype.constructor = hxl.classes.RowFilter;
1121 
1122 /**
1123  * Override HXLFIlter.iterator to return only select rows.
1124  *
1125  * @return an iterator object that will skip rows that fail to pass at
1126  * least one of the predicates.
1127  */
1128 hxl.classes.RowFilter.prototype.iterator = function() {
1129     var iterator = this.source.iterator();
1130     var outer = this;
1131     return {
1132         next: function() {
1133             var row;
1134             while (row = iterator.next()) {
1135                 if (outer._tryPredicates(row)) {
1136                     return row;
1137                 }
1138             }
1139             return null;
1140         }
1141     }
1142 }
1143 
1144 /**
1145  * Operator functions.
1146  */
1147 hxl.classes.RowFilter.OPERATORS = {
1148     '=': function (a, b) { return a == b; },
1149     '!=': function (a, b) { return a != b; },
1150     '<': function (a, b) { return a < b; },
1151     '<=': function (a, b) { return a <= b; },
1152     '>': function (a, b) { return a > b; },
1153     '>=': function (a, b) { return a >= b; },
1154     '~': function (a, b) { return a.match(b); },
1155     '!~': function (a, b) { return !a.match(b); }
1156 };
1157 
1158 /**
1159  * Precompile the tag patterns in the predicates.
1160  *
1161  * @param predicates the predicates as input
1162  * @return a compiled/normalised list of predicates
1163  * @exception if one of the tag patterns is malformed
1164  */
1165 hxl.classes.RowFilter.prototype._compilePredicates = function(predicates) {
1166 
1167     /**
1168      * Helper function: compile the tag pattern, if present
1169      */
1170     var parsePattern = function (pattern) {
1171         if (pattern) {
1172             return hxl.classes.Pattern.parse(pattern);
1173         } else {
1174             return null;
1175         }
1176     };
1177 
1178     /**
1179      * Helper function: if test is a plain string, create an equality function.
1180      */
1181     var parseTest = function (test) {
1182         if (typeof(test) != 'function') {
1183             test = hxl.norm(test);
1184             return function (value) { return hxl.norm(value) == test; };
1185         } else {
1186             return test;
1187         }
1188     };
1189 
1190     /**
1191      * Helper function: parse a string predicate.
1192      */
1193     var parsePredicate = function (s) {
1194         var operator, expected;
1195 
1196         // loose expression (parsing the pattern will verify)
1197         var result = s.match(/^\s*([^!=~<>]+)\s*(!?[=~]|<=?|>=?)(.*)$/);
1198         if (result) {
1199             operator = hxl.classes.RowFilter.OPERATORS[result[2]];
1200             expected = hxl.norm(result[3]);
1201             return {
1202                 pattern: parsePattern(result[1]),
1203                 test: function (value) { return operator(hxl.norm(value), expected); }
1204             };
1205         } else {
1206             throw Error("Bad predicate: " + s);
1207         }
1208     };
1209 
1210     // If it's not a list, wrap it in one
1211     if (!Array.isArray(predicates)) {
1212         predicates = [ predicates ];
1213     }
1214 
1215     // Map the list to a compiled/normalised version
1216     return predicates.map(function (predicate) {
1217         if (typeof(predicate) == 'object') {
1218             return {
1219                 pattern: parsePattern(predicate.pattern),
1220                 test: parseTest(predicate.test)
1221             };
1222         } else if (typeof(predicate) == 'function') {
1223             return {
1224                 test: predicate
1225             };
1226         } else {
1227             return parsePredicate(predicate);
1228         }
1229     });
1230 }
1231 
1232 /**
1233  * Return success if _any_ of the predicates succeeds.
1234  */
1235 hxl.classes.RowFilter.prototype._tryPredicates = function(row) {
1236     var predicate;
1237 
1238     // Try every predicate on the row
1239     for (var i = 0; i < this.predicates.length; i++) {
1240         predicate = this.predicates[i];
1241 
1242         // If the first part is set, then it's a tag pattern
1243         // test only the values with hashtags that match
1244         if (predicate.pattern) {
1245             var values = row.getAll(predicate.pattern);
1246             for (var j = 0; j < values.length; j++) {
1247                 if (predicate.test(values[i])) {
1248                     return !this.invert;
1249                 }
1250             }
1251         } 
1252 
1253         // If the first part is not set, then it's a row predicate
1254         // test the whole row at once
1255         else {
1256             if (this.invert) {
1257                 return !predicate.test(row);
1258             } else {
1259                 return predicate.test(row);
1260             }
1261         }
1262     }
1263 
1264     return this.invert;
1265 }

1289 /**
1290  * Override hxl.classes.BaseFilter.getColumns to return only the allowed columns.
1291  *
1292  * This method triggers lazy processing that also saves the indices for
1293  * slicing the data itself.
1294  *
1295  * @return a list of hxl.classes.Column objects.
1296  */
1297 hxl.classes.ColumnFilter.prototype.getColumns = function() {
1298     if (typeof(this._savedColumns) == 'undefined') {
1299 
1300         /**
1301          * Check if any of the patterns matches the column.
1302          */
1303         var columnMatches = function(column, patterns) {
1304             for (var i = 0; i < patterns.length; i++) {
1305                 if (patterns[i].match(column)) {
1306                     return true;
1307                 }
1308             }
1309             return false;
1310         };
1311 
1312         // we haven't extracted the columns before, so do it now
1313         var newColumns = [];
1314         var newIndices = [];
1315 
1316         // check every column against the patterns
1317         for (var i = 0; i < this.source.columns.length; i++) {
1318             var is_match = columnMatches(this.source.columns[i], this.patterns);
1319             if ((is_match && !this.invert) || (!is_match && this.invert)) {
1320                 newColumns.push(this.source.columns[i]);
1321                 newIndices.push(i);
1322             }
1323         }
1324 
1325         // save the columns and indices for future use
1326         this._savedColumns = newColumns;
1327         this._savedIndices = newIndices;
1328     }
1329 
1330     // return the saved columns
1331     return this._savedColumns;
1332 }

1346             var row = iterator.next();
1347             if (row) {
1348                 // Use the saved indices to slice values
1349                 values = [];
1350                 for (i = 0; i < outer._savedIndices.length; i++) {
1351                     values.push(row.values[outer._savedIndices[i]]);
1352                 }
1353                 return new hxl.classes.Row(values, columns);
1354             } else {
1355                 // end of data
1356                 return null;
1357             }
1358         }
1359     }
1360 }

1414 /**
1415  * Override hxl.classes.BaseFilter.getColumns to return only the columns for the aggregation report.
1416  *
1417  * Will list the tags that match the patterns provided in the
1418  * constructor, as well as a #meta+count tag, and aggregation tags if
1419  * the aggregation parameter was included.
1420  *
1421  * @return a list of hxl.classes.Column objects
1422  */
1423 hxl.classes.CountFilter.prototype.getColumns = function() {
1424     var cols, indices, tagspec;
1425     if (!this._savedColumns) {
1426         cols = [];
1427         indices = [];
1428         for (var i = 0; i < this.source.columns.length; i++) {
1429             for (var j = 0; j < this.patterns.length; j++) {
1430                 if (this.patterns[j].match(this.source.columns[i])) {
1431                     cols.push(this.source.columns[i]);
1432                     indices.push(i);
1433                     break;
1434                 }
1435             }
1436         }
1437         cols.push(hxl.classes.Column.parse('#meta+count'));
1438         if (this.aggregate) {
1439             tagspec = this.aggregate.tag;
1440             cols.push(hxl.classes.Column.parse(tagspec + '+sum'));
1441             cols.push(hxl.classes.Column.parse(tagspec + '+avg'));
1442             cols.push(hxl.classes.Column.parse(tagspec + '+min'));
1443             cols.push(hxl.classes.Column.parse(tagspec + '+max'));
1444         }
1445         this._savedColumns = cols;
1446         this._savedIndices = indices;
1447     }
1448     return this._savedColumns;
1449 }

1451 /**
1452  * Override hxl.classes.BaseFilter.iterator to return a set of rows with aggregated values.
1453  *
1454  * Each row represents a unique set of values and the number of times
1455  * it occurs.
1456  *
1457  * @return an iterator over the aggregated data.
1458  */
1459 hxl.classes.CountFilter.prototype.iterator = function() {
1460     var columns = this.columns; // will trigger lazy column creation
1461     var data = this._aggregateData();
1462     var pos = 0;
1463     return {
1464         next: function () {
1465             if (pos < data.length) {
1466                 return new hxl.classes.Row(data[pos++], columns);
1467             } else {
1468                 return null;
1469             }
1470         }
1471     };
1472 }

1474 /**
1475  * Monster ugly function to aggregate data.
1476  * FIXME: can I decompose this into smaller parts?
1477  */
1478 hxl.classes.CountFilter.prototype._aggregateData = function() {
1479     var row, key, values, value, entry, aggregates;
1480     var data_map = {};
1481     var aggregate_map = {};
1482     var data = [];
1483     var iterator = this.source.iterator();
1484 
1485     // Make a unique map of data values
1486     while (row = iterator.next()) {
1487         key = this._makeKey(row);
1488 
1489         // Always do a count
1490         if (data_map[key]) {
1491             data_map[key] += 1;
1492         } else {
1493             data_map[key] = 1;
1494         }
1495 
1496         // Aggregate numeric values if requested
1497         if (this.aggregate) {
1498             // try parsing, and proceed only if it's numeric
1499             value = parseFloat(row.get(this.aggregate));
1500             if (!isNaN(value)) {
1501                 entry = aggregate_map[key];
1502                 if (entry) {
1503                     // Not the first value
1504                     entry.total++;
1505                     entry.sum += value;
1506                     entry.avg = (entry.avg * (entry.total - 1) + value) / entry.total;
1507                     entry.min = (value < entry.min ? value : entry.min);
1508                     entry.max = (value > entry.max ? value : entry.max);
1509                 } else {
1510                     // the first value
1511                     aggregate_map[key] = {
1512                         total: 1,
1513                         sum: value,
1514                         avg: value,
1515                         min: value,
1516                         max: value
1517                     }
1518                 }
1519             }
1520         }
1521     }
1522 
1523     // Generate the data from the map
1524     for (key in data_map) {
1525 
1526         // Retrieve the values from the key
1527         values = key.split("\0");
1528 
1529         // Add the count
1530         values.push(data_map[key]);
1531 
1532         // Add other aggregates if requested
1533         if (this.aggregate) {
1534             entry = aggregate_map[key];
1535             if (entry) {
1536                 values = values.concat([
1537                     entry.sum,
1538                     entry.avg,
1539                     entry.min,
1540                     entry.max
1541                 ]);
1542             } else {
1543                 values = values.concat([
1544                     '', '', '', ''
1545                 ]);
1546             }
1547         }
1548 
1549         // Row is finished.
1550         data.push(values);
1551     }
1552 
1553     // return all the rows
1554     return data;
1555 }

1557 /**
1558  * Construct a unique key from the requested values in a row of data.
1559  *
1560  * @return the unique key as a single string.
1561  */
1562 hxl.classes.CountFilter.prototype._makeKey = function(row) {
1563     var i, index;
1564     var values = [];
1565     for (i = 0; i < this._savedIndices.length; i++) {
1566         values.push(row.values[this._savedIndices[i]]);
1567     }
1568     return values.join("\0");
1569 }

1577  * HXL filter to rename a column (new header and tag).
1578  *

1581  * @param pattern the tag pattern to replace

1585  * @param index the zero-based index of the match to replace. If
1586  * undefined or null or false, replace *all* matches.

1590     this.pattern = hxl.classes.Pattern.parse(pattern);

1592     this.newHeader = newHeader;
1593     this.index = index;

1600 /**
1601  * Get the renamed columns.
1602  */
1603 hxl.classes.RenameFilter.prototype.getColumns = function() {
1604     var cols, header;
1605     var pattern = this.pattern;
1606     var tagspec = this.newTagspec;
1607     var index = this.index;
1608     if (this._savedColumns === undefined) {
1609         cols = [];
1610         // loop through the columns, translating as needed
1611         this.source.getColumns().forEach(function (col) {
1612             // Index has to be 0 or undefined for a match
1613             if (pattern.match(col) && !index) {
1614                 // we have a match!
1615                 if (this.header === undefined) {
1616                     header = col.header;
1617                 } else {
1618                     header = this.header;
1619                 }
1620                 col = hxl.classes.Column.parse(tagspec, header);
1621                 cols.push(col);
1622             } else {
1623                 // no match: use existing column
1624                 cols.push(col);
1625             }
1626             // decrement the index counter if present
1627             if (index) {
1628                 index -= 1;
1629             }
1630         });
1631         this._savedColumns = cols;
1632     }
1633     return this._savedColumns;
1634 }

1709 /**
1710  * Add index attributes (+i0, +i1, etc.) to a repeated tag.
1711  *
1712  * This is useful for query-type processing, where it's not otherwise

1715  * that's not available, so you can automatically number the tags
1716  * as #org+i0, #org+i1, etc., from left to right.
1717  *
1718  * @constructor
1719  * @this{IndexFilter}
1720  * @param {hxl.classes.Source} source the hxl.classes.Source
1721  * @param {string} pattern the tag pattern to replace (see {@link hxl.classes.Pattern}).
1722  */
1723 hxl.classes.IndexFilter = function (source, pattern) {
1724     hxl.classes.BaseFilter.call(this, source);
1725     this.pattern = hxl.classes.Pattern.parse(pattern);
1726 }

1729 hxl.classes.IndexFilter.prototype.constructor = hxl.classes.IndexFilter;

1731 hxl.classes.IndexFilter.prototype.getColumns = function() {
1732     var pattern = this.pattern;
1733     if (this._savedColumns == undefined) {
1734         var i = 0;
1735         var cols = [];
1736         this.source.columns.forEach(function (col) {
1737             if (pattern.match(col)) {
1738                 col = col.clone();
1739                 col.attributes.push('i' + i++);
1740             }
